Example

Example usage of Expense Model

const { ObjectId } = mongoose.Schema.Types;
const user = new User({...}); // see User Model
const travel = new Travel({...}); // see Travel Model
const currency = new Currency({...}); // see * Currency Model

const expenseObject = {
  travel: travel._id, // link to travel
  type: 'Flight',
  description: 'Tel Aviv - Barcelona - Tel Aviv',
  date: new Date('2019-11-07'),
  currency: 'EUR',
  curRate: currency._id, // link to currency
  amount: 89,
  amountConverted: 100,
  _user: user._id // link to user
};
const expense = new Expense(expenseObject);
expense.save();
