const path = require('path');

const {
  createLogger, transports, format, Container, ...rest
} = require('winston');

const env = process.env.NODE_ENV || 'development';

const getFileData = () => {
  const mainModule = path.basename(process.mainModule.filename);
  const filenameSplit = __filename.split('\\');
  const filename = filenameSplit[filenameSplit.length - 1];
  const data = {
    mainModule,
    filepath: __filename,
    filename,
    filenameSplit
  };
  return data;
};

const printf = format.printf(info => {
  let {
    level, message, label: lbl, metadata, ...rest
  } = info;
  let { label, requestId } = metadata;
  const { filename } = getFileData();
  label = !label ? 'main' : label;
  requestId = requestId === undefined ? 100 : requestId;
  const res = (`[${level}] [${requestId}] [${label}] ${message}\n`).toString();
  return res;
});

const logger = createLogger({
  level: env === 'development' ? 'debug' : 'info',
  format: format.combine(
    format.json(),
    format.splat(),
    format.prettyPrint(),
    format.colorize({ all: true }),
    format.timestamp(),
    format.metadata({
      fillExcept: ['message', 'level', 'timestamp'],
      fillWith: ['label', 'requestId']
    }),
    printf
  ),
  defaultMeta: { service: 'user-service' },
  transports: [new transports.Console()],
  exitOnError: false
});

class Logger {
  constructor(label = 'child', requestId = 999) {
    this._filedata = getFileData;
    Object.freeze(this._filedata);
    this.__logger = logger;
    this._label = label;
    this._requestId = requestId;
    this._logger = this.__logger.child({
      label: `${this._label}`,
      requestId: this._requestId
    });
    this._childLoggers = [];
    Logger.count++;
  }

  get mainLogger() {
    return this.__logger;
  }

  /**
   * Property default logger
   *
   * @readonly
   * @memberof Logger
   */
  get logger() {
    return this._logger;
  }

  /**
   * This logger label
   * @memberof Logger
   */
  get label() {
    return this._label;
  }
